```python
def get_klines(self, limit: int = 50) -> List[Dict]:
    """获取 K 线数据"""
    try:
        response = requests.get(
            f"{BASE_URL}/api/binance/klines",
            params={
                'symbol': self.symbol,
                'interval': '1m',
                'limit': limit
            },
            timeout=10
        )
        data = response.json()
        
        if data.get('success'):
            return data.get('klines', [])
        return []
    except Exception as e:
        print(f"❌ 获取 K 线失败：{e}")
        return []
```

## 🛠️ 如果需要本地存储

### 推荐实现（SQLite）

**表结构**:
```sql
-- K 线数据表
CREATE TABLE klines (
    id INTEGER PRIMARY KEY,
    symbol TEXT,
    timestamp INTEGER,
    open REAL,
    high REAL,
    low REAL,
    close REAL,
    volume REAL,
    UNIQUE(symbol, timestamp)
);

-- 指标数据表
CREATE TABLE indicators (
    id INTEGER PRIMARY KEY,
    symbol TEXT,
    timestamp INTEGER,
    rsi REAL,
    macd REAL,
    signal REAL,
    UNIQUE(symbol, timestamp)
);

-- 创建索引
CREATE INDEX idx_symbol_time ON klines(symbol, timestamp);
CREATE INDEX idx_ind_symbol_time ON indicators(symbol, timestamp);
```

**代码示例**:
```python
import sqlite3

class LocalStorage:
    def __init__(self, db_path='data/market.db'):
        self.conn = sqlite3.connect(db_path)
        self.create_tables()
    
    def save_klines(self, symbol, klines):
        """保存 K 线数据"""
        for k in klines:
            self.conn.execute(
                'INSERT OR REPLACE INTO klines VALUES (?,?,?,?,?,?,?,?)',
                (None, symbol, k['time'], k['open'], k['high'], 
                 k['low'], k['close'], k['volume'])
            )
        self.conn.commit()
    
    def get_klines(self, symbol, limit=50):
        """获取 K 线数据"""
        cursor = self.conn.execute(
            'SELECT * FROM klines WHERE symbol=? ORDER BY timestamp DESC LIMIT ?',
            (symbol, limit)
        )
        return cursor.fetchall()
```
