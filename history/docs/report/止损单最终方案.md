```python
# api/stop_loss_manager.py

class StopLossManager:
    """止损管理器（本地监控）"""
    
    def __init__(self):
        self.stop_orders = {}  # 存储所有止损单
    
    def set_stop_loss(self, symbol: str, side: str, quantity: float, 
                     stop_price: float, strategy_id: int):
        """设置止损单"""
        self.stop_orders[strategy_id] = {
            'symbol': symbol,
            'side': side,  # long/short
            'quantity': quantity,
            'stop_price': stop_price,
            'active': True
        }
        print(f"✅ 止损单已设置：{symbol} {side.upper()} @ {stop_price}")
    
    def check_all_stops(self, current_prices: dict):
        """检查所有止损单"""
        triggered = []
        
        for strategy_id, stop_info in self.stop_orders.items():
            if not stop_info['active']:
                continue
            
            symbol = stop_info['symbol']
            current_price = current_prices.get(symbol, 0)
            
            if current_price <= 0:
                continue
            
            # 多单止损检查
            if stop_info['side'] == 'long' and current_price <= stop_info['stop_price']:
                triggered.append({
                    'strategy_id': strategy_id,
                    'symbol': symbol,
                    'side': 'SELL',
                    'quantity': stop_info['quantity'],
                    'trigger_price': current_price,
                    'stop_price': stop_info['stop_price']
                })
                print(f"🔴 止损触发：{symbol} 多单 @ {current_price} (止损价：{stop_info['stop_price']})")
            
            # 空单止损检查
            elif stop_info['side'] == 'short' and current_price >= stop_info['stop_price']:
                triggered.append({
                    'strategy_id': strategy_id,
                    'symbol': symbol,
                    'side': 'BUY',
                    'quantity': stop_info['quantity'],
                    'trigger_price': current_price,
                    'stop_price': stop_info['stop_price']
                })
                print(f"🔴 止损触发：{symbol} 空单 @ {current_price} (止损价：{stop_info['stop_price']})")
        
        return triggered
    
    def remove_stop(self, strategy_id: int):
        """移除止损单（策略平仓后）"""
        if strategy_id in self.stop_orders:
            del self.stop_orders[strategy_id]
            print(f"✅ 止损单已移除：策略{strategy_id}")
```

### 集成到网关

```python
# gateway.py

# 全局止损管理器
stop_loss_manager = StopLossManager()

# 定时任务（每 30 秒检查一次）
def check_stop_losses():
    """定时检查止损"""
    # 获取当前价格
    prices = {}
    for symbol in ['BTCUSDT', 'ETHUSDT', 'LINKUSDT', 'AVAXUSDT']:
        price = get_current_price(symbol)
        prices[symbol] = price
    
    # 检查止损
    triggered = stop_loss_manager.check_all_stops(prices)
    
    # 执行止损平仓
    for trigger in triggered:
        client = BinanceClient(api_key, secret_key, testnet=True)
        result = client.place_futures_order(
            symbol=trigger['symbol'],
            side=trigger['side'],
            type='MARKET',
            quantity=trigger['quantity']
        )
        
        if result.get('success'):
            print(f"✅ 止损已执行：{trigger['symbol']} @ {result['price']}")
            stop_loss_manager.remove_stop(trigger['strategy_id'])
        else:
            print(f"❌ 止损执行失败：{result.get('error')}")

# 添加到 cron 或定时任务
import schedule
schedule.every(30).seconds.do(check_stop_losses)
```
