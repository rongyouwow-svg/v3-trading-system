### 长期修复：改进启动脚本

**新启动脚本：**
```bash
#!/bin/bash
# 启动网关服务（带进程检查）

LOG_FILE="/home/admin/.openclaw/workspace/quant/logs/gateway_start.log"

log() {
    echo "$(date '+%Y-%m-%d %H:%M:%S'): $1" >> "$LOG_FILE"
}

# 1. 检查是否已有进程在运行
if pgrep -f "gateway.py" > /dev/null; then
    log "⚠️ 检测到网关进程已在运行"
    
    # 2. 停止旧进程
    log "🔄 停止旧进程..."
    pkill -9 -f "gateway.py"
    sleep 3
    
    # 3. 验证已停止
    if pgrep -f "gateway.py" > /dev/null; then
        log "🔴 无法停止旧进程，需要人工干预"
        exit 1
    fi
    log "✅ 旧进程已停止"
fi

# 4. 启动新进程
log "🦞 启动网关服务..."
cd /home/admin/.openclaw/workspace/quant
nohup /home/admin/.pyenv/versions/3.10.0/bin/python3 gateway.py > /tmp/gateway.log 2>&1 &

# 5. 验证启动成功
sleep 10
if curl -s http://localhost:8080/api/health | grep -q "ok"; then
    log "✅ 网关启动成功"
else
    log "🔴 网关启动失败，查看日志：/tmp/gateway.log"
    exit 1
fi
```
