# api/stop_loss_manager.py

class StopLossManager:
    """止损管理器（支持测试网/实盘分离）"""
    
    def __init__(self, testnet=True):
        self.testnet = testnet
        self.stop_orders = {}  # 测试网模拟止损
        self.real_orders = {}  # 实盘实际止损单 ID
    
    def set_stop_loss(self, symbol: str, side: str, quantity: float, 
                     stop_price: float, strategy_id: int) -> Dict:
        """
        设置止损单
        
        Returns:
            {'success': True, 'order_id': '...', 'testnet': True/False}
        """
        if self.testnet:
            # 测试网：模拟止损（仅记录）
            self.stop_orders[strategy_id] = {
                'symbol': symbol,
                'side': side,
                'quantity': quantity,
                'stop_price': stop_price,
                'active': True
            }
            return {
                'success': True,
                'order_id': f'testnet_{strategy_id}',
                'testnet': True,
                'message': '测试网模拟止损（不实际下单）'
            }
        else:
            # 实盘：调用币安 Algo Order API
            result = self._place_algo_stop_order(symbol, side, quantity, stop_price)
            if result.get('success'):
                self.real_orders[strategy_id] = result['order_id']
            return result
    
    def _place_algo_stop_order(self, symbol: str, side: str, 
                               quantity: float, stop_price: float) -> Dict:
        """实盘：调用币安 Algo Order API"""

            'stopPrice': stop_price,
            'quantity': quantity,
            'reduceOnly': 'true',
            'priceProtect': 'true',
            'newOrderRespType': 'RESULT'
        }
        
        # 调用 API...
        # 返回结果
    
    def check_testnet_stops(self, current_prices: dict):
        """测试网专用：检查模拟止损"""
        if not self.testnet:
            return []
        
        triggered = []
        for strategy_id, stop_info in self.stop_orders.items():
            if not stop_info['active']:
                continue
            
            symbol = stop_info['symbol']
            current_price = current_prices.get(symbol, 0)
            
            # 多单止损
            if stop_info['side'] == 'long' and current_price <= stop_info['stop_price']:
                triggered.append({
                    'strategy_id': strategy_id,
                    'symbol': symbol,
                    'side': 'SELL',
                    'quantity': stop_info['quantity'],
                    'trigger_price': current_price
                })
                print(f"🔴 [测试网] 止损触发：{symbol} @ {current_price}")
        
        return triggered
