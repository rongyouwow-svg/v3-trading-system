2. **创建策略执行器**
```python
# api/auto_sim_executor.py

class AutoSimExecutor:
    def __init__(self):
        self.simulators = {}  # 每个币种一个模拟器
    
    def start_strategy(self, symbol, leverage, trade_amount):
        """启动策略"""
        simulator = AutoTradingSimulator(
            symbol=symbol,
            initial_capital=trade_amount * leverage
        )
        self.simulators[symbol] = simulator
    
    def execute_cycle(self, symbol, current_price):
        """执行一个循环"""
        if symbol not in self.simulators:
            return
        
        simulator = self.simulators[symbol]
        signal = simulator.generate_signal(current_price)
        
        # 根据信号执行
        if signal['type'] == 'OPEN':
            # 开仓
            client.place_futures_order(...)
        elif signal['type'] == 'ADD':
            # 加仓
            client.place_futures_order(...)
        elif signal['type'] == 'CLOSE':
            # 平仓
            client.place_futures_order(...)
```

3. **添加定时任务**
```python
# gateway.py

import schedule
import time

auto_sim_executor = AutoSimExecutor()

# 每分钟执行一次
schedule.every(1).minutes.do(execute_auto_sim_cycles)

def execute_auto_sim_cycles():
    """每分钟执行 auto_sim 循环"""
    for symbol, simulator in auto_sim_executor.simulators.items():
        price = get_current_price(symbol)
        auto_sim_executor.execute_cycle(symbol, price)

# 后台运行定时任务
import threading
def run_schedule():
    while True:
        schedule.run_pending()
        time.sleep(1)

threading.Thread(target=run_schedule, daemon=True).start()
```
