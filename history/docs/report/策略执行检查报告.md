### 修复 2：添加委托单显示

```html
<div id="open-orders">
    <h3>📋 委托单（含止损）</h3>
    <div id="orders-list">加载中...</div>
</div>

<script>
async function loadOpenOrders() {
    const resp = await fetch(`${API_BASE}/binance/orders?symbol=ETHUSDT`);
    const data = await resp.json();
    
    if (data.orders && data.orders.length > 0) {
        document.getElementById('orders-list').innerHTML = data.orders.map(o => `
            <div class="order-item">
                ${o.side} | ${o.type} | ${o.origQty} @ $${o.price}
                ${o.stopPrice ? `| 触发价：$${o.stopPrice}` : ''}
                | 状态：${o.status}
            </div>
        `).join('');
    } else {
        document.getElementById('orders-list').innerHTML = '<div class="empty">无委托单</div>';
    }
}
</script>
```
