### 1.1 问题统计

根据系统记录，之前遇到的主要问题：

| 问题类型 | 出现频率 | 影响 |
|---------|---------|------|
| 代码格式错误 | 高 | 进程崩溃 |
| 模块间数据格式不一致 | 高 | 功能异常 |
| 异常未捕获 | 高 | 进程崩溃 |

| 精度处理错误 | 中 | 订单失败 |

### 1.2 典型案例

**案例 1: 代码格式错误**


## 2. 第一标准：统一数据格式规范

### 2.1 统一返回结果类

**文件**: `modules/utils/result.py`

```python
from typing import Dict, Any, Optional
from dataclasses import dataclass
from enum import Enum
from datetime import datetime


class StatusCode(Enum):
    """状态码"""
    SUCCESS = "success"
    ERROR = "error"
    WARNING = "warning"


@dataclass
class Result:
    """统一返回结果 - 所有模块必须使用"""
    
    code: StatusCode
    message: str
    data: Optional[Dict[str, Any]] = None
    error_code: Optional[str] = None
    
    def to_dict(self) -> Dict:
        """转换为字典"""
        result = {
            'code': self.code.value,
            'message': self.message,
            'timestamp': datetime.now().isoformat()
        }
        if self.data is not None:
            result['data'] = self.data
        if self.error_code is not None:
            result['error_code'] = self.error_code
        return result
    
    @classmethod
    def ok(cls, data: Dict = None, message: str = "操作成功") -> 'Result':
        """成功结果"""
        return cls(code=StatusCode.SUCCESS, message=message, data=data)
    
    @classmethod
    def fail(cls, error_code: str, message: str, data: Dict = None) -> 'Result':
        """失败结果"""
        return cls(code=StatusCode.ERROR, message=message, data=data, error_code=error_code)
    
    @property
    def is_success(self) -> bool:
        """是否成功"""
        return self.code == StatusCode.SUCCESS
```

## 3. 第二标准：精度处理规范

### 3.1 禁止使用 float 处理金额

**文件**: `modules/utils/precision.py`

```python
from decimal import Decimal, ROUND_DOWN
from typing import Tuple


class PrecisionUtils:
    """精度处理工具 - 所有模块必须使用"""
    
    # 精度映射（实际应从交易所获取）
    STEP_SIZE_MAP = {
        'ETHUSDT': Decimal('0.001'),
        'BTCUSDT': Decimal('0.001'),
        'AVAXUSDT': Decimal('0.1'),
        'LINKUSDT': Decimal('0.1'),
        'UNIUSDT': Decimal('0.1'),
    }
    
    TICK_SIZE_MAP = {
        'ETHUSDT': Decimal('0.01'),
        'BTCUSDT': Decimal('0.1'),
        'AVAXUSDT': Decimal('0.001'),
        'LINKUSDT': Decimal('0.001'),
        'UNIUSDT': Decimal('0.001'),
    }
    
    @classmethod
    def get_step_size(cls, symbol: str) -> Decimal:
        """获取数量精度"""
        return cls.STEP_SIZE_MAP.get(symbol, Decimal('0.001'))
    
    @classmethod
    def get_tick_size(cls, symbol: str) -> Decimal:
        """获取价格精度"""
        return cls.TICK_SIZE_MAP.get(symbol, Decimal('0.01'))
    
    @classmethod
    def normalize_quantity(cls, symbol: str, quantity: Decimal) -> Decimal:
        """标准化数量（向下取整到 stepSize）"""
        step_size = cls.get_step_size(symbol)
        return quantity.quantize(step_size, rounding=ROUND_DOWN)
    
    @classmethod
    def normalize_price(cls, symbol: str, price: Decimal) -> Decimal:
        """标准化价格（向下取整到 tickSize）"""
        tick_size = cls.get_tick_size(symbol)
        return price.quantize(tick_size, rounding=ROUND_DOWN)
    
    @classmethod
    def validate_quantity(cls, symbol: str, quantity: Decimal) -> Tuple[bool, str]:
        """验证数量是否合法"""
        step_size = cls.get_step_size(symbol)
        
        if quantity <= 0:
            return False, "数量必须大于 0"
        
        if quantity < step_size:
            return False, f"数量不能小于 {step_size}"
        
        normalized = cls.normalize_quantity(symbol, quantity)
        if normalized != quantity:
            return False, f"数量精度错误，应该是 {normalized}"
        
        return True, ""
    
    @classmethod
    def validate_price(cls, symbol: str, price: Decimal) -> Tuple[bool, str]:
        """验证价格是否合法"""
        tick_size = cls.get_tick_size(symbol)
        
        if price <= 0:
            return False, "价格必须大于 0"
        
        normalized = cls.normalize_price(symbol, price)
        if normalized != price:
            return False, f"价格精度错误，应该是 {normalized}"
        
        return True, ""
```

### 4.2 异常处理装饰器

**文件**: `modules/utils/decorators.py`

```python
from functools import wraps
from typing import Any, Callable
import traceback
import logging

from .result import Result

logger = logging.getLogger(__name__)


def handle_exceptions(default_return: Any = None):
    """
    异常处理装饰器 - 所有对外函数必须使用
    
    用法:
        @handle_exceptions()
        def create_order(...):
            pass
        
        @handle_exceptions(default_return=Result.fail("DEFAULT", "默认返回"))
        def get_price(...):
            pass
    """
    def decorator(func: Callable):
        @wraps(func)
        def wrapper(*args, **kwargs):
            try:
                return func(*args, **kwargs)
            except TradingException as e:
                # 业务异常
                logger.error(f"{func.__name__} 业务异常：{e.message}", exc_info=True)
                return Result.fail(error_code=e.error_code, message=e.message)
            except Exception as e:
                # 系统异常
                error_msg = f"{func.__name__} 异常：{str(e)}\n{traceback.format_exc()}"
                logger.error(error_msg)
                
                if default_return is not None:
                    return default_return
                return Result.fail(
                    error_code="INTERNAL_ERROR",
                    message=f"{func.__name__} 执行失败：{str(e)}"
                )
        return wrapper
    return decorator


def retry_on_exception(max_retries: int = 3, delay: float = 1.0):
    """
    重试装饰器 - 用于网络请求等可重试操作
    
    用法:
        @retry_on_exception(max_retries=3, delay=1.0)
        def place_order(...):
            pass
    """
    def decorator(func: Callable):
        @wraps(func)
        def wrapper(*args, **kwargs):
            last_exception = None
            for attempt in range(max_retries):
                try:
                    return func(*args, **kwargs)
                except NetworkException as e:
                    last_exception = e
                    if attempt < max_retries - 1:
                        logger.warning(f"{func.__name__} 重试 {attempt + 1}/{max_retries}: {e.message}")
                        import time
                        time.sleep(delay * (2 ** attempt))  # 指数退避
                    continue
                except Exception as e:
                    # 非网络异常不重试
                    raise
            
            if last_exception:
                raise last_exception
        return wrapper
    return decorator
```

### 6.1 日志配置

```python
import logging
import json
import sys
from datetime import datetime
from typing import Optional, Dict, Any


class JSONFormatter(logging.Formatter):
    """JSON 格式日志 - 所有模块必须使用"""
    
    def format(self, record: logging.LogRecord) -> str:
        log_data: Dict[str, Any] = {
            'timestamp': datetime.now().isoformat(),
            'level': record.levelname,
            'module': record.module,
            'function': record.funcName,
            'line': record.lineno,
            'message': record.getMessage(),
        }
        
        # 添加异常信息
        if record.exc_info:
            log_data['exception'] = self.formatException(record.exc_info)
        
        # 添加额外数据
        if hasattr(record, 'extra_data'):
            log_data['data'] = record.extra_data
        
        return json.dumps(log_data, ensure_ascii=False)


def setup_logger(
    name: str,
    level: int = logging.INFO,
    log_file: Optional[str] = None,
    console_output: bool = True
) -> logging.Logger:
    """
    设置日志
    
    Args:
        name: 日志名称
        level: 日志级别
        log_file: 日志文件路径
        console_output: 是否输出到控制台
    
    Returns:
        logging.Logger: 日志对象
    """
    logger = logging.getLogger(name)
    logger.setLevel(level)
    
    # 清除已有处理器
    logger.handlers.clear()
    
    # JSON 格式化
    formatter = JSONFormatter()
    
    # 文件处理器
    if log_file:
        file_handler = logging.FileHandler(log_file, encoding='utf-8')
        file_handler.setFormatter(formatter)
        logger.addHandler(file_handler)
    
    # 控制台处理器
    if console_output:
        console_handler = logging.StreamHandler(sys.stdout)
        console_handler.setFormatter(formatter)
        logger.addHandler(console_handler)
    
    return logger


# 使用示例
logger = setup_logger('strategy_engine', log_file='logs/strategy.log')
logger.info("策略已启动", extra={'extra_data': {'symbol': 'ETHUSDT'}})
logger.error("订单创建失败", extra={'extra_data': {'error': '余额不足'}})
```

```python
from pydantic import BaseModel, validator, Field
from decimal import Decimal
from typing import Optional, Literal


class OrderCreateRequest(BaseModel):
    """订单创建请求 - 自动验证"""
    
    symbol: str = Field(..., min_length=5, pattern='^[A-Z]+USDT$')
    side: Literal['BUY', 'SELL']
    type: Literal['MARKET', 'LIMIT', 'STOP_MARKET', 'STOP_LIMIT']
    quantity: Decimal = Field(..., gt=0, decimal_places=8)
    price: Optional[Decimal] = Field(None, gt=0, decimal_places=8)
    leverage: int = Field(default=1, ge=1, le=125)
    reduce_only: bool = Field(default=False)
    
    @validator('quantity')
    def validate_quantity_precision(cls, v):
        """验证数量精度"""
        if v.as_tuple().exponent < -8:
            raise ValueError("数量精度不能超过 8 位小数")
        return v
    
    @validator('price')
    def validate_price_precision(cls, v):
        """验证价格精度"""
        if v and v.as_tuple().exponent < -8:
            raise ValueError("价格精度不能超过 8 位小数")
        return v


class StrategyStartRequest(BaseModel):
    """策略启动请求"""
    
    symbol: str = Field(..., min_length=5, pattern='^[A-Z]+USDT$')
    strategy_id: str = Field(..., min_length=1)
    side: Literal['LONG', 'SHORT']
    leverage: int = Field(..., ge=1, le=125)
    amount: Decimal = Field(..., gt=0)
    params: dict = Field(default_factory=dict)


class StopLossCreateRequest(BaseModel):
    """止损单创建请求"""
    
    symbol: str = Field(..., min_length=5, pattern='^[A-Z]+USDT$')
    side: Literal['BUY', 'SELL']
    trigger_price: Decimal = Field(..., gt=0)
    quantity: Decimal = Field(..., gt=0)
    stop_price: Optional[Decimal] = None
```

```python
from cryptography.fernet import Fernet
import base64
import os


class SecretManager:
    """敏感信息管理器"""
    
    def __init__(self, key: Optional[bytes] = None):
        if key is None:
            key = Fernet.generate_key()
        self.fernet = Fernet(key)
        self.key = key
    
    def encrypt(self, plaintext: str) -> str:
        """加密"""
        return self.fernet.encrypt(plaintext.encode()).decode()
    
    def decrypt(self, ciphertext: str) -> str:
        """解密"""
        return self.fernet.decrypt(ciphertext.encode()).decode()
    
    def save_key(self, path: str):
        """保存密钥"""
        with open(path, 'wb') as f:
            f.write(self.key)
    
    @classmethod
    def load_key(cls, path: str) -> 'SecretManager':
        """加载密钥"""
        with open(path, 'rb') as f:
            key = f.read()
        return cls(key=key)
```

### 13.1 缓存使用

**文件**: `modules/utils/cache.py`

```python
from datetime import datetime, timedelta
from typing import Optional, Any, Dict
from decimal import Decimal


class PriceCache:
    """价格缓存"""
    
    def __init__(self, ttl_seconds: int = 60):
        self.cache: Dict[str, tuple] = {}
        self.ttl = timedelta(seconds=ttl_seconds)
    
    def get(self, symbol: str) -> Optional[Decimal]:
        """获取缓存价格"""
        if symbol in self.cache:
            data, timestamp = self.cache[symbol]
            if datetime.now() - timestamp < self.ttl:
                return data
            del self.cache[symbol]
        return None
    
    def set(self, symbol: str, price: Decimal):
        """设置缓存价格"""
        self.cache[symbol] = (price, datetime.now())
    
    def clear(self):
        """清空缓存"""
        self.cache.clear()
```

```python
import asyncio
from aiohttp import ClientSession


class AsyncConnector:
    """异步连接器"""
    
    def __init__(self):
        self.session: Optional[ClientSession] = None
    
    async def __aenter__(self):
        self.session = ClientSession()
        return self
    
    async def __aexit__(self, exc_type, exc_val, exc_tb):
        await self.session.close()
    
    async def get_price(self, symbol: str) -> Decimal:
        """异步获取价格"""
        url = f"https://api.binance.com/futures/v1/ticker/price?symbol={symbol}"
        async with self.session.get(url) as response:
            data = await response.json()
            return Decimal(data['price'])
```
