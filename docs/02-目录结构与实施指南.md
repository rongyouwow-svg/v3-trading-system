```python
import json

class StrategyManager:
    """策略管理器（支持热插拔）"""
    
    def __init__(self):
        self.strategies: Dict[str, Any] = {}  # {symbol: 策略实例}
        self.threads: Dict[str, threading.Thread] = {}  # {symbol: thread}
        self.stop_flags: Dict[str, bool] = {}  # {symbol: stop_flag}
        
        # 持久化文件
        self.persistence_file = '/root/.openclaw/workspace/quant/data/plugin_strategies.json'
        
        # 加载已有策略
        self._load_strategies()
    
    def _load_strategies(self):
        """从持久化文件加载策略"""
        if os.path.exists(self.persistence_file):
            try:
                with open(self.persistence_file, 'r', encoding='utf-8') as f:
                    data = json.load(f)
                
                for symbol, strategy_data in data.items():
                    if strategy_data.get('status') == 'running':
                        print(f"🔌 [热插拔] 发现运行中策略：{symbol} - {strategy_data.get('strategy_id')}")
                        # TODO: 恢复策略实例
                        
            except Exception as e:
                print(f"⚠️ 加载策略文件失败：{e}")
    
    def start_strategy(self, symbol: str, strategy_id: str, **kwargs) -> Dict:
        """启动策略"""
        # 检查是否已有策略
        if symbol in self.strategies:
            return {
                'success': False,
                'error': f'{symbol} 已有活跃策略'
            }
        
        # 加载策略
        strategy = self._load_strategy(strategy_id, **kwargs)
        if not strategy:
            return {
                'success': False,
                'error': f'未知策略：{strategy_id}'
            }
        
        # 保存策略实例
        self.strategies[symbol] = strategy
        
        # 启动信号计算线程
        stop_flag = False
        self.stop_flags[symbol] = stop_flag
        
        thread = threading.Thread(
            target=self._signal_calculation_loop,
            args=(symbol, stop_flag),
            daemon=True
        )
        thread.start()
        self.threads[symbol] = thread
        
        # 持久化
        self._save_strategy(symbol, strategy_id, kwargs)
        
        return {
            'success': True,
            'message': f'{symbol} 策略已启动',
            'strategy': strategy_id
        }
    
    def stop_strategy(self, symbol: str) -> Dict:
        """停止策略"""
        if symbol not in self.strategies:
            return {
                'success': False,
                'error': f'{symbol} 没有活跃策略'
            }
        
        # 设置停止标志
        self.stop_flags[symbol] = True
        
        # 等待线程退出
        if symbol in self.threads:
            self.threads[symbol].join(timeout=10)
            del self.threads[symbol]
        
        # 移除策略
        del self.strategies[symbol]
        
        # 持久化
        self._remove_strategy(symbol)
        
        return {
            'success': True,
            'message': f'{symbol} 策略已停止'
```

```python
                        'leverage': strategy_data.get('leverage', 1),
                        'amount': strategy_data.get('amount', 0),
                        'start_time': strategy_data.get('start_time'),
                        'is_restored': True  # 标记为恢复的策略
                    }
                    
        except Exception as e:
            print(f"⚠️ 加载策略文件失败：{e}")
```

## 📝 四、常见问题

### Q1: 策略显示"未知策略"

**原因**: `plugin_strategies.json` 和 `strategy_engine` 数据不同步

**解决**: 在 `load_strategies()` 中兼容读取 `plugin_strategies.json`

### Q2: 交易记录空白

**原因**: 缺少 `/api/trades` 路由

**解决**: 添加交易记录查询 API

### Q3: 网关重启后策略丢失

**原因**: 策略状态未持久化

**解决**: 确保 `plugin_strategies.json` 实时更新，启动时自动加载
